def performance_metric2(y_true, y_predict):
    """"""
    yt = np.array(y_true)
    yp = np.array(y_predict)
    yt_mean = yt.mean()
    
    sse = np.array((yt-yp)**2).sum()
    sst = np.array((yt-yt_mean)**2).sum()
    score = 1 - sse*1.0/sst
    return score

def cross_val_score(estimator, X, y, scoring=performance_metric, cv=3):
    """  """
    X_list = np.array(X).tolist()
    y_list = np.array(y).tolist()

    # [[],]
    # hard_code3
    X_cv = []
    X1 = X_list[:len(X_list)/cv]
    X2 = X_list[len(X_list)/cv:2*len(X_list)/cv]
    X3 = X_list[2*len(X_list)/cv:]
    X_cv.append(X1)
    X_cv.append(X2)
    X_cv.append(X3)

    y_cv = []
    y1 = y_list[:len(y_list)/cv]
    y2 = y_list[len(y_list)/cv:2*len(y_list)/cv]
    y3 = y_list[2*len(y_list)/cv:]
    y_cv.append(y1)
    y_cv.append(y2)
    y_cv.append(y3)

    scores = []
    i = 0
    while i < cv:
        X_test = X_cv[i]
        X_train = []
        for _ in X_cv:
            if _ != X_test:
                X_train += _

        y_test = y_cv[i]
        y_train = []
        for _ in y_cv:
            if _ != y_test:
                y_train += _

        # fit[[],]
        estimator.fit(X_train, y_train)
        y_pred = estimator.predict(X_test)
        score = scoring(y_test, y_pred)
        scores.append(score)
        i += 1

    # 
    return scores

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeRegressor
from sklearn.model_selection import KFold, GridSearchCV
from sklearn.metrics import make_scorer
from sklearn.metrics import r2_score

# load data
data = pd.read_csv('bj_housing.csv')
prices = data['Value']
features = data.drop('Value', axis=1)

# gen train-test data
X_train, X_test, y_train, y_test = train_test_split(features, prices, test_size=0.2)


# model, cv, evaluate
def fit_model(X, y):
    # KFold()3
    cross_validator = KFold()
    # model
    regressor = DecisionTreeRegressor()
    # params
    params = {'max_depth': range(1, 11)}
    # metric func
    scoring_fnc = make_scorer(performance_metric)

    grid = GridSearchCV(regressor, params, scoring=scoring_fnc, cv=cross_validator)
    grid = grid.fit(X, y)
    # print pd.DataFrame(grid.cv_results_)
    # 
    return grid.best_estimator_


# performance_metric: r2_score
def performance_metric(y_true, y_predict):
    """"""
    score = r2_score(y_true, y_predict)
    return score
