import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    if random.random() < epsilon: # 
        action = random.choice(actions) # 
    else: 
        action = sorted(qline.items(), key=lambda x:(-x[1], x[0]))[0][0] #  Q 
    return action
